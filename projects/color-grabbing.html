1   视觉系统——多色HSV识别

OpenCV多色HSV空间色块检测
import cv2
import numpy as np

# 四种颜色HSV阈值范围
color_ranges = {
    'red':   ([0, 100, 100],   [10, 255, 255]),
    'green': ([35, 100, 100],  [85, 255, 255]),
    'blue':  ([100, 100, 100], [130, 255, 255]),
    'yellow':([20, 100, 100],  [35, 255, 255]),
}

def detect_blocks(frame):
    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
    results = []

    for color_name, (lower, upper) in color_ranges.items():
        mask = cv2.inRange(hsv, np.array(lower),
                           np.array(upper))
        # 形态学处理: 开运算去噪 + 闭运算填充
        kernel = cv2.getStructuringElement(
            cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)

        # 轮廓检测
        contours, _ = cv2.findContours(
            mask, cv2.RETR_EXTERNAL,
            cv2.CHAIN_APPROX_SIMPLE)

        for cnt in contours:
            area = cv2.contourArea(cnt)
            if area > 500:  # 最小面积过滤
                x, y, w, h = cv2.boundingRect(cnt)
                cx, cy = x + w//2, y + h//2
                results.append({
                    'color': color_name,
                    'center': (cx, cy),
                    'area': area,
                    'bbox': (x, y, w, h)
                })
    return results

2   视觉伺服闭环控制

将色块中心偏差作为PID输入，控制底盘运动趋近目标：

视觉伺服PID闭环底盘控制
class VisualServoPID:
    def __init__(self):
        self.Kp_x = 0.003   # 水平偏差比例增益
        self.Kp_d = 0.005   # 距离比例增益
        self.target_cx = 320 # 图像中心x (640/2)
        self.target_area = 5000 # 目标面积 (对应期望距离)

    def compute(self, cx, area):
        # 水平偏差 -> 旋转速度
        error_x = cx - self.target_cx
        omega = -self.Kp_x * error_x

        # 面积偏差 -> 前进速度
        error_d = self.target_area - area
        velocity = self.Kp_d * error_d

        return velocity, omega

3   六步抓取流程

步骤动作控制方式
1视觉检测目标色块OpenCV HSV分割+轮廓检测
2像素→世界坐标相机标定+坐标映射
3底盘PID趋近视觉伺服闭环驱动
4精细位姿对准小步微调，消除残差
5机械臂夹取舵机控制夹爪闭合
6搬运至目标区域导航到放置点→释放
