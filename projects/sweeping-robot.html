3   编码器与航迹推算

通过TIM2/TIM5编码器模式采集轮速，实现航迹推算（Dead Reckoning）：

Δ s = (Δ sL + Δ sR)/(2), Δθ = (Δ sR - Δ sL)/(D)
xk+1 = xk + Δ s · cosθk + (Δθ)/(2), yk+1 = yk + Δ s · sinθk + (Δθ)/(2)

其中 D 为两轮轴距，Δ sL、Δ sR 分别为左右轮弧长增量。结合JY901S IMU的航向角进行加权融合修正累积漂移。

4   PID闭环速度控制

增量式PID速度控制器（50ms采样周期）
typedef struct {
    float Kp, Ki, Kd;       // Kp=0.1, Ki=0.1, Kd=0.001
    float error, last_error, prev_error;
    float output;
    float out_max, out_min;
} PID_TypeDef;

float PID_Incremental(PID_TypeDef *pid, float target, float actual) {
    pid->error = target - actual;

    // 增量式PID: delta_u = Kp*(e-e1) + Ki*e + Kd*(e-2e1+e2)
    float delta = pid->Kp * (pid->error - pid->last_error)
                + pid->Ki * pid->error
                + pid->Kd * (pid->error - 2*pid->last_error
                             + pid->prev_error);

    pid->output += delta;

    // 输出限幅，防止PWM溢出
    if (pid->output > pid->out_max) pid->output = pid->out_max;
    if (pid->output < pid->out_min) pid->output = pid->out_min;

    pid->prev_error = pid->last_error;
    pid->last_error = pid->error;

    return pid->output;
}

5   JY901S IMU数据解析

JY901S串口数据帧解析（11字节帧）
// JY901S 数据帧格式: 0x55 + TYPE + 8字节数据 + SUM
void IMU_ParseFrame(uint8_t *buf) {
    if (buf[0] != 0x55) return;

    uint8_t type = buf[1];
    switch (type) {
        case 0x53:  // 欧拉角数据帧
            // Roll/Pitch/Yaw = raw * (180.0 / 32768.0)
            imu_data.roll  = ((int16_t)(buf[3]<<8|buf[2]))
                             / 32768.0f * 180.0f;
            imu_data.pitch = ((int16_t)(buf[5]<<8|buf[4]))
                             / 32768.0f * 180.0f;
            imu_data.yaw   = ((int16_t)(buf[7]<<8|buf[6]))
                             / 32768.0f * 180.0f;
            break;
        case 0x51:  // 加速度数据帧
            imu_data.ax = ((int16_t)(buf[3]<<8|buf[2]))
                          / 32768.0f * 16.0f * 9.8f;
            imu_data.ay = ((int16_t)(buf[5]<<8|buf[4]))
                          / 32768.0f * 16.0f * 9.8f;
            break;
    }
}

// 14字节下行帧: 0xBB + 编码器(4B) + IMU(4B) + 传感器(4B) + SUM
void SendStatusFrame(void) {
    uint8_t frame[14];
    frame[0] = 0xBB;  // 帧头
    // 编码器脉冲数 (左/右)
    frame[1] = (encoder_left >> 8) & 0xFF;
    frame[2] = encoder_left & 0xFF;
    frame[3] = (encoder_right >> 8) & 0xFF;
    frame[4] = encoder_right & 0xFF;
    // IMU 航向角
    int16_t yaw_raw = (int16_t)(imu_data.yaw * 100);
    frame[5] = (yaw_raw >> 8) & 0xFF;
    frame[6] = yaw_raw & 0xFF;
    // 传感器状态 (碰撞+悬崖)
    frame[7] = collision_left | (collision_right << 1);
    frame[8] = cliff_sensor[0] | (cliff_sensor[1] << 1)
             | (cliff_sensor[2] << 2) | (cliff_sensor[3] << 3);
    // 校验和
    uint8_t sum = 0;
    for (int i = 0; i < 13; i++) sum += frame[i];
    frame[13] = sum;
    HAL_UART_Transmit(&huart_bt, frame, 14, 100);
}

8   上位机——PyQt5实时可视化

PyQt5上位机轨迹可视化核心
class TrajectoryWidget(QWidget):
    def __init__(self):
        super().__init__()
        self.fig, self.ax = plt.subplots(figsize=(6, 6))
        self.canvas = FigureCanvas(self.fig)
        self.trajectory_x = []
        self.trajectory_y = []

    def update_position(self, x, y, theta):
        """接收下位机解算的位姿，实时绘制轨迹"""
        self.trajectory_x.append(x)
        self.trajectory_y.append(y)
        self.ax.clear()
        self.ax.plot(self.trajectory_x, self.trajectory_y,
                     'b-', linewidth=1.5, label='轨迹')
        # 绘制当前朝向箭头
        dx = 0.05 * math.cos(math.radians(theta))
        dy = 0.05 * math.sin(math.radians(theta))
        self.ax.arrow(x, y, dx, dy, head_width=0.02,
                      color='red')
        self.ax.set_aspect('equal')
        self.ax.legend()
        self.canvas.draw()
