电机驱动系统设计——从有刷到无刷

薛雅文 2024–2025

本项目围绕电机驱动系统展开，涵盖有刷直流电机开环驱动、无刷直流电机（BLDC）六步换相驱动 以及 PI 闭环控制器的设计与实现。硬件端基于 STM32F103C8T6 进行固件开发， 配套设计了两版 Altium Designer PCB 并制板焊接。 项目完整覆盖了电机驱动从原理推导、固件编写到 PCB 制作的全流程。

关键词：STM32, 有刷电机, BLDC, 六步换相, H 桥, PI 控制, PCB 设计

1   有刷直流电机驱动

1.1   硬件设计

采用分立 MOSFET 搭建 H 桥：上桥臂（Q1, Q2）由 TIM2/TIM3 的 PWM 输出驱动， 下桥臂（Q3, Q4）由 GPIO 开关控制。对角导通实现正反转。

参数配置
MCUSTM32F103C8T6（72 MHz，HSE 8 MHz × PLL ×9）
PWM 频率30 kHz（Timer Period = 2399, Prescaler = 0）
PWM 通道TIM2_CH4 (PB11), TIM3_CH4 (PB1)
占空比范围100–2300（步进 239，对应 ~4%–96%）
main.c — H 桥电机正反转控制
void Motor_Stop(void) {
    HAL_TIM_PWM_Stop(&htim2, TIM_CHANNEL_4);
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_SET);
    HAL_TIM_PWM_Stop(&htim3, TIM_CHANNEL_4);
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10, GPIO_PIN_SET);
}

void Motor_UpdateOutput(void) {
    if (Motor_Dir == 1) {   // 正转：Q2(TIM2_CH4) PWM, Q3(PB0) LOW
        HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_4);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_4, Motor_Speed);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_RESET);
        HAL_TIM_PWM_Stop(&htim3, TIM_CHANNEL_4);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10, GPIO_PIN_SET);
    } else {                // 反转：Q1(TIM3_CH4) PWM, Q4(PB10) LOW
        HAL_TIM_PWM_Stop(&htim2, TIM_CHANNEL_4);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_SET);
        HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_4);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_4, Motor_Speed);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10, GPIO_PIN_RESET);
    }
}

1.2   电流与电压检测

电流检测采用低端检流电阻 Rshunt = 0.01 Ω，经 20 倍运放放大后送入 ADC1_CH5。 ADC 参考电压 1.2 V，电流计算公式：

I = (Vadc − 1.2) × 5    [A]

母线电压通过 33 kΩ / 10 kΩ 分压后接入 ADC1_CH6，比例系数 ×4.3。 电流与电压实时显示在 SSD1306 OLED（128×64，软件模拟 I²C）上。

main.c — ADC 电压/电流采集（16 次均值滤波）
#define VOL_RATIO   4.3f     // R16=33k, R22=10k → Vin = Vadc×4.3
#define CUR_VREF    1.2f     // 电流检测参考电压
#define CUR_GAIN    5.0f     // I = (Vout - Vref) / 0.2
#define ADC_AVG_N   16

static void ADC_Read_CurVol_Avg(uint16_t *raw_cur, uint16_t *raw_vol) {
    uint32_t sum_cur = 0, sum_vol = 0;
    for (int n = 0; n < ADC_AVG_N; n++) {
        HAL_ADC_Start(&hadc1);
        if (HAL_ADC_PollForConversion(&hadc1, 10) == HAL_OK)
            sum_cur += HAL_ADC_GetValue(&hadc1);     // CH5 = 电流
        if (HAL_ADC_PollForConversion(&hadc1, 10) == HAL_OK)
            sum_vol += HAL_ADC_GetValue(&hadc1);     // CH6 = 电压
        HAL_ADC_Stop(&hadc1);
    }
    *raw_cur = (uint16_t)(sum_cur / ADC_AVG_N);
    *raw_vol = (uint16_t)(sum_vol / ADC_AVG_N);
}

static void Measure_Update(void) {
    ADC_Read_CurVol_Avg(&adc_raw_cur, &adc_raw_vol);
    float v_cur = (adc_raw_cur * 3.3f) / 4095.0f;
    float v_vol = (adc_raw_vol * 3.3f) / 4095.0f;
    meas_voltage_V = v_vol * VOL_RATIO;
    meas_current_A = (v_cur - CUR_VREF) * CUR_GAIN;
    if (meas_current_A < 0.0f) meas_current_A = 0.0f;
}
main.c — OLED 实时显示电压/电流/PWM
static void OLED_ShowMeas(uint16_t speed, uint8_t dir,
                          float vin, float ia) {
    char line[24];
    OLED_Fill(0);
    snprintf(line, 24, "PWM:%4d", speed);          OLED_Puts(0,7, line);
    snprintf(line, 24, "DIR:%1d",  dir?1:0);       OLED_Puts(0,5, line);
    snprintf(line, 24, "V:%5d", (int)(vin*1000));  OLED_Puts(0,3, line);
    snprintf(line, 24, "I:%5d", (int)(ia*1000));   OLED_Puts(0,1, line);
    OLED_Update();
}

1.3   用户交互

4 个按键（PB4–PB7）分别控制：停止、正反转切换、加速、减速。 开环控制模式下，按键直接修改 PWM 占空比。

2   无刷直流电机（BLDC）驱动

2.1   六步换相原理

三相 BLDC 通过 Hall 传感器确定转子位置。3 个 Hall 信号（PA8–PA10）编码为 3-bit 状态（1–6）， 对应六步换相表中的一个状态。每个状态导通两相（一相高侧 PWM + 一相低侧 GPIO）， 第三相悬空。

Hall 状态导通相高侧 PWM低侧 GPIO
1 (001)U→VHI_U (TIM3_CH4)LO_V (PA6)
2 (010)W→VHI_W (TIM2_CH4)LO_V (PA6)
3 (011)W→UHI_W (TIM2_CH4)LO_U (PB0)
4 (100)V→UHI_V (TIM3_CH2)LO_U (PB0)
5 (101)V→WHI_V (TIM3_CH2)LO_W (PA2)
6 (110)U→WHI_U (TIM3_CH4)LO_W (PA2)

PWM 频率同为 30 kHz，固定占空比约 83%（CCR = 2000 / ARR = 2399）。

main.c — BLDC 六步换相（Hall → switch-case）
BLDC_STATE = 0;
if (HAL_GPIO_ReadPin(GPIOA, HALL_U_Pin)==GPIO_PIN_SET) BLDC_STATE |= 0x01;
if (HAL_GPIO_ReadPin(GPIOA, HALL_V_Pin)==GPIO_PIN_SET) BLDC_STATE |= 0x02;
if (HAL_GPIO_ReadPin(GPIOA, HALL_W_Pin)==GPIO_PIN_SET) BLDC_STATE |= 0x04;

switch (BLDC_STATE) {
    case 1:  // U 上桥 PWM, W 下桥导通
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_4, SET_PWM);
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_2, 0);
        __HAL_TIM_SetCompare(&htim2, TIM_CHANNEL_4, 0);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0,  GPIO_PIN_SET);    // U- 关
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_6,  GPIO_PIN_SET);    // V- 关
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2,  GPIO_PIN_RESET);  // W- 开
        break;
    case 2:  // W 上桥 PWM, V 下桥导通
        __HAL_TIM_SetCompare(&htim2, TIM_CHANNEL_4, SET_PWM);
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_4, 0);
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_2, 0);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0,  GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_6,  GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2,  GPIO_PIN_SET);
        break;
    /* case 3–6: 类似模式, 依次导通 W→U, V→U, V→W, U→W */
    default: // 全关保护
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_4, 0);
        __HAL_TIM_SetCompare(&htim3, TIM_CHANNEL_2, 0);
        __HAL_TIM_SetCompare(&htim2, TIM_CHANNEL_4, 0);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_6, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2, GPIO_PIN_SET);
        break;
}

2.2   电流检测与滤波

母线电流经 ADC1_CH5 采集，采用 16 次均值滤波降低噪声。 由于运放输出存在零点偏移（实测 1.14 V），需在软件中减去偏移量后换算为实际电流值。 电压检测通过 100 kΩ / 10 kΩ 分压（比例 ×11），经 ADC1_CH4 读取。 采样周期 20 ms。

3   PI 速度闭环控制

3.1   控制器设计

增量式 PI 控制器参数：Kp = 0.8，采样周期 T = 2 ms，Ki = Kp / T = 400。 控制律为：

VPI = Kp · e[k] + VI
VI = VI + Ki · Tn · e[k]

其中 e[k] 为目标转速与实际转速的偏差，Tn 为实际积分时间步长。 积分项具有累积效果，可消除稳态误差。

main.c — PI 速度闭环控制器
float Kp = 0.8f;
float T  = 0.002f;              // 采样周期 2ms
float Ki = Kp / T;              // Ki = 400
float VI = 0.0f;                // 积分累积项

void PI_Control(float Vref, float Vin, float dt) {
    float Ve  = Vin - Vref;     // 误差 = 实测 - 目标
    VI += Ki * dt * Ve;         // 积分项累加
    float Vpi = Kp * Ve + VI;   // PI 输出
    // 限幅后写入 PWM
    if (Vpi > 2399) Vpi = 2399;
    if (Vpi < 0)    Vpi = 0;
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_4, (uint16_t)Vpi);
}

4.2   第二版（MOTOR_260305）

修订版在初版基础上大幅扩展电路规模：新增 4 个 IC（电流检测运放与栅极驱动）、 23 个电阻、7 个电容与 1 个保护二极管。PCB 从头重新布局， 走线密度显著增加。新增元件主要用于：

  • 额外的稳压/去耦电容
  • 栅极驱动电路（MOSFET 栅极电荷快充/泄放）
  • ESD/TVS 保护
